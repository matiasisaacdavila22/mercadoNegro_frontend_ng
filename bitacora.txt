**definimos las rutas en app-routing.mosule.ts***

const routes: Routes = [
  {
    path: 'home',
    component:DefaultComponent
  },
  {
    path: '',
    pathMatch: 'full',
    redirectTo: '/home'
  },
  **se agrego a posteriori del enrutamiento a security**
    {
    path: '/security',
    loadChildren: () => import('./modules/security/security.module')
     .then(m => m.SecurityModule)
  },
    {
    path: 'parameters',
    loadChildren: () => import('./modules/parameters/parameters.module')
     .then(m => m.ParametersModule)
  },
   /**this must be the last one*/
  {
    path:'**',
    redirectTo: '/home'
  }
];
